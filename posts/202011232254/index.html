/**
 * @param {number[]} A
 * @return {number}
 */
var maxScoreSightseeingPair = function(A) {
    let count = A[0] + A[1] - 1 
    for (let i = 0; i<A.length; i++) {
        for (let j = i + 1; j < A.length; j++) {
            if (A[i] + A[j] + i -j > count) {
                count = A[i] + A[j] + i -j
            }
        }
    }
    return count
};
