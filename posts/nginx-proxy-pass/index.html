Nginx Proxy Pass

2020-11-09 — 1 min read 
  server {
    listen       80;
    server_name  cloudwave.cn;

    location /productlist {
      # 匹配所有以/productlist开头所有页面都被拦截
      # 将匹配到的页面请求转发到新的静态资源服务器
      proxy_pass http://new.cloudwave.cn;
    }
  }
  server {
    listen       443;
    server_name  cloudwave.cn;

    location /productlist {
      proxy_pass https://new.cloudwave.cn;
    }
  }
