  1. php 文件代码
<?php
namespace Cli\Controller;
use Think\Controller;

class TaskController extends Controller {

    public function _initialize()
    {
        // 判断执行环境
        if(PHP_SAPI != 'cli'){
            exit('deny!');
        }
    }

    function exec(){
        $hour = date('H');  // 小时

        if(intval($hour) == 0) {
            D('YybStatistics')->increase([]);  // 自动进行今日统计
        }
    }
}
